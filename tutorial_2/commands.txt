connect to mysql db
=> mysql -u root -proot

check databases
=> show databases;

create database with name 'tutorial'
=> create database tutorial;

enter into the database
=> use tutorial;

list tables in a database
=> show tables;

Create table students
=> create table students(
	name varchar(25),
	roll int,
	score int,
	grade varchar(1),
	address varchar(50)
);

Describe table
=> desc students;


Save student in table - 85, 70, 60, 50, 40

Read student from table
=> select * from students;
=> select * from students where name = 'Rajesh';

Update student in table
=> update students
set score=59,grade='D' 
where name = 'Rajesh';


Delete student from table
=> delete from students where score < 40;
